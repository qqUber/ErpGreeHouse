<tasks>

<task type="auto">
  <name>Task 1: Start the FastAPI application server in background</name>
  <files>middleware/app/main.py</files>
  <action>
    Run the FastAPI application server using uvicorn in the background.
    Use the command: `cd middleware && uvicorn app.main:app --host 0.0.0.0 --port 8000 > server.log 2>&1 & echo $! > server.pid`
    Wait for the server to start by checking the /health endpoint with retries.
  </action>
  <verify>
    cd middleware && for i in {1..10}; do sleep 2; curl -s -o /dev/null -w "%{http_code}" http://localhost:8000/health | grep -q "200" && break; done && curl -s -o /dev/null -w "%{http_code}" http://localhost:8000/health
  </verify>
  <done>FastAPI application server is running on port 8000</done>
</task>

<task type="auto">
  <name>Task 2: Verify application endpoints</name>
  <files>middleware/app/main.py</files>
  <action>
    Test all available API endpoints to ensure they return valid responses.
    Check endpoints: /health, /api/v1/auth/me, /api/v1/loyalty/tiers
  </action>
  <verify>
    cd middleware && curl -s -w "%{http_code}" http://localhost:8000/health | grep -E "200" && curl -s -w "%{http_code}" http://localhost:8000/api/v1/auth/me | grep -E "401" && curl -s -w "%{http_code}" http://localhost:8000/api/v1/loyalty/tiers | grep -E "200"
  </verify>
  <done>All tested endpoints are accessible and returning valid responses</done>
</task>

<task type="auto">
  <name>Task 3: Stop the application server</name>
  <files>middleware/app/main.py</files>
  <action>
    Stop the running uvicorn server using the PID file.
    Command: `cd middleware && if [ -f server.pid ]; then kill $(cat server.pid) 2>/dev/null || true; rm -f server.pid; fi`
  </action>
  <verify>
    cd middleware && curl -s -o /dev/null -w "%{http_code}" http://localhost:8000/health 2>/dev/null || echo "Server stopped"
  </verify>
  <done>Application server has been stopped</done>
</task>

</tasks>
